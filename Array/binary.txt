####################문제####################

숫자를 입력 받아 이진수로 출력하는 프로그램을 작성하시오.

###########################################


#include <stdio.h>

int main()
{
    int number;
    int binary[20] = { 0, };
    
    scanf("%d", &number);

    int position = 0;
    while (1)
    {
        binary[position]=number % 2;    // 2로 나누었을 때 나머지를 배열에 저장
        number=number / 2;             // 2로 나눈 몫을 저장

        position++;    // 자릿수 변경

        if (number == 0)    // 몫이 0이 되면 반복을 끝냄
            break;
    }

    // 배열의 요소를 역순으로 출력
    for (int i=position-1; i>=0; i--)
    {
        printf("%d", binary[i]);
    }

    printf("\n");

    return 0;
}
